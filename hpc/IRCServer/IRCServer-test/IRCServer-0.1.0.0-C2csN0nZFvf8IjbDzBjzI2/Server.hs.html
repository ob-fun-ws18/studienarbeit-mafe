   37 -- |An event that should be handled by the main server
   38 data Event = ClientMsg User Msg -- ^A parsed message from a client
   39         | NewUser User Handle   -- ^A new user can bea reached via the given handle
   40     deriving (Show, Eq)
   41 
   42 -- |This type maps from the nicknames to the handles to the client.
   43 type Users = (Map.Map String Handle)
   44 -- |This type maps channel names to the nicknames of the users in the channel.
   45 type Channels = (Map.Map String [String])
   46 
   47 -- |This function sets up the listening socket, starts the main server, and starts
   48 -- a loop for accepting connections.
   49 startServer :: PortNumber -- ^ The port that should accept new connections
   50             -> IO ()
   51 startServer port = do
   52     sock <- socket AF_INET Stream 0
   53     setSocketOption sock ReuseAddr 1
   54     bind sock (SockAddrInet port iNADDR_ANY)
   55     listen sock 2
   56     chan <- newChan
   57     forkIO (mainServer chan Map.empty Map.empty)
   58     acceptConnections sock chan
   59 
   60 -- |This function accepts new connections from clients and starts the communication
   61 -- with them in a new thread.
   62 acceptConnections :: Socket     -- ^The listening socket
   63                   -> Chan Event -- ^The channel to communicate with the main server
   64                   -> IO ()
   65 acceptConnections sock chan = do
   66     conn <- accept sock
   67     forkIO (startConn conn chan)
   68     acceptConnections sock chan

   70 -- |The main server waits for events that the clients generate. It also keeps track
   71 -- of the connected users and the channels. It modifies the states and sends replys
   72 -- to the clients.
   73 mainServer :: Chan Event -- ^The channels to receive events from the clients
   74            -> Users      -- ^The currently connected users with their handles
   75            -> Channels   -- ^The currently available channels with their users
   76            -> IO ()
   77 mainServer chan users channels = do
   78     event <- readChan chan
   79     print event
   80     (new_users, new_channels) <- handleEvent users channels event
   81     mainServer chan new_users new_channels
   82 
   83 -- |This function contains the main logic of how to react to the different events.
   84 -- It processes the event, alters the users and the channel information where neccessary
   85 -- and sends responses to the clients.
   86 handleEvent :: Users                -- ^The current users
   87             -> Channels             -- ^The current channel information
   88             -> Event                -- ^The event received from the client
   89             -> IO (Users, Channels) -- ^It returns the new state of users and channels
   90 handleEvent users channels (NewUser user@(FullUser nick username name) hdl) = do
   91     let new_users = Map.insert nick hdl users
   92     hPutStrLn hdl (":localhost 001 " ++ nick ++ " :Welcome to the Internet Relay Network " ++ toString user)
   93     return (new_users, channels)
   94 handleEvent users channels (ClientMsg user@(FullUser nick _ _) (Ping server)) = do
   95     sendToUser users nick "PONG localhost"
   96     return (users, channels)
   97 handleEvent users channels (ClientMsg user@(FullUser nick _ _) msg@(PrivMsg receiver _)) = do
   98     if head receiver == '#'
   99         then do
  100             let receivers = filter (nick /=) (Map.findWithDefault [] receiver channels)
  101             sendToAllUsers users receivers (buildMsg user msg)
  102         else sendToUser users receiver $ buildMsg user msg
  103     return (users, channels)
  104 handleEvent users channels (ClientMsg user@(FullUser nick _ _) msg@(Join channel)) = do
  105     let members = nick : Map.findWithDefault [] channel channels
  106         new_channels = Map.insert channel members channels
  107     sendToAllUsers users members $ buildMsg user msg
  108     return (users, new_channels)
  109 handleEvent users channels (ClientMsg _ (JoinMulti [])) = return (users, channels)
  110 handleEvent users channels (ClientMsg user msg@(JoinMulti (c : cs))) = do
  111     (new_users, new_channels) <- handleEvent users channels (ClientMsg user (Join c))
  112     handleEvent new_users new_channels (ClientMsg user (JoinMulti cs))
  113 handleEvent users channels (ClientMsg user@(FullUser nick _ _) msg@(Part channel _)) = do
  114     let members = Map.findWithDefault [] channel channels
  115     sendToAllUsers users members $ buildMsg user msg
  116     let new_channels = Map.insert channel (filter (nick /=) members) channels
  117     return (users, new_channels)
  118 handleEvent users channels (ClientMsg user@(FullUser nick _ _) msg@(Quit _)) = do
  119     let hdl = users ! nick
  120         new_users = Map.delete nick users
  121         new_channels = Map.map (filter (nick /=)) channels
  122     hClose hdl
  123     sendToAllUsers users (Map.keys new_users) (buildMsg user msg)
  124     return (new_users , new_channels)
  125 handleEvent users channels (ClientMsg user@(FullUser nick _ _) msg@(IsOn nicks)) = do
  126     let usersOn = filter (`Map.member` users) nicks
  127     sendToUser users nick $ buildMsg user (IsOn usersOn)
  128     return (users, channels)
  129 handleEvent users channels _ = return (users, channels)

  131 -- |This function sends a message to a specified user.
  132 sendToUser :: Users  -- ^The map of users to look up the correct handle
  133            -> String -- ^The nickname of the user to send to
  134            -> String -- ^The message to send
  135            -> IO ()
  136 sendToUser users nick = hPutStrLn (users ! nick)
  137 
  138 -- |This function sends the same message to many users.
  139 sendToAllUsers :: Users    -- ^The map of users to look up the correct handles
  140                -> [String] -- ^The nicknames of the users to send to
  141                -> String   -- ^The message to send
  142                -> IO ()
  143 sendToAllUsers _ [] _ = return ()
  144 sendToAllUsers users receivers msg = do
  145     sendToUser users (head receivers) msg
  146     sendToAllUsers users (tail receivers) msg

  148 -- |This function sets up the connection to a client and starts the loop that handles
  149 -- the further communication.
  150 startConn :: (Socket, SockAddr) -- ^The socket information for the client
  151           -> Chan Event         -- ^The channel to communictate with the main server
  152           -> IO ()
  153 startConn (sock, _) chan = do
  154     hdl <- socketToHandle sock ReadWriteMode
  155     hSetBuffering hdl NoBuffering
  156     loopConn hdl chan NoUser
  157 
  158 -- |This function handles the communication from the client to the server. It reads
  159 -- the messages, parses them, and acts accordingly.
  160 loopConn :: Handle     -- ^The handle to receive data from the client
  161          -> Chan Event -- ^The channel to communicate with the main server
  162          -> User       -- ^The data of the user that is handled by this connection
  163          -> IO ()
  164 loopConn hdl chan user = do
  165     eof <- hIsEOF hdl
  166     if eof then
  167         writeChan chan $ ClientMsg user $ Quit "Client lost connection."
  168     else do
  169         line <- hGetLine hdl
  170         let msg = parseMsg $ init line
  171             (new_user, event) = handleMsg hdl user msg
  172         putStrLn ("\n" ++ line)
  173         print msg
  174         Data.Foldable.forM_ event (writeChan chan)
  175         unless (isFinalMsg msg) (loopConn hdl chan new_user)
  176 
  177 isFinalMsg :: Msg -> Bool
  178 isFinalMsg (Quit _) = True
  179 isFinalMsg _ = False
  180 
  181 handleMsg :: Handle -> User -> Msg -> (User, Maybe Event)
  182 handleMsg _ NoUser (Nick nick) = (NickUser nick, Nothing)
  183 handleMsg hdl (NickUser nick) (User username name) = (user, Just (NewUser user hdl))
  184         where user = FullUser nick username name
  185 handleMsg _ user msg = (user, Just (ClientMsg user msg))
