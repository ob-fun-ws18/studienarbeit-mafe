    3 Description: Message parser

    5 This module contains functions to parse and build Messages.

    7 module Client (
    8       Msg(..)
    9     , parseMsg
   10     , buildMsg
   11     , parseCommand
   12     ) where
   13 
   14 import Text.Regex.PCRE
   15 import Data.List.Split
   16 import User
   17 
   18 -- |Messages that can be received from the client
   19 data Msg = Plain String
   20         | Nick String                   -- ^Nickname message with the new nick
   21         | User String String            -- ^User message with username and full name
   22         | PrivMsg String String         -- ^Private message with receiver and content
   23         | Join String                   -- ^Join message with the channel
   24         | JoinMulti [String]            -- ^Join message with multiple channels
   25         | Part String String            -- ^Part message with the channel name and message
   26         | Ping String                   -- ^Ping message
   27         | Quit String                   -- ^Quit with message
   28         | IsOn [String]                 -- ^Is on message with the nick names
   29     deriving (Show, Eq)
   30 
   31 -- |This function builds the messages for the client.
   32 buildMsg :: User        -- ^The sender of the message
   33          -> Msg         -- ^The message
   34          -> String      -- ^Message to send to the client
   35 buildMsg user (PrivMsg receiver msg) = ":" ++ toString user ++ " PRIVMSG " ++ receiver ++ " :" ++ msg
   36 buildMsg user (Join channel) = ":" ++ toString user ++ " JOIN " ++ channel
   37 buildMsg user (Part channel msg) = ":" ++ toString user ++ " PART " ++ channel ++ " :" ++ msg
   38 buildMsg user (Quit msg) = ":" ++ toString user ++ " QUIT :" ++ msg
   39 buildMsg (FullUser nick _ _) (IsOn users) = ":localhost 303 " ++ nick ++ " :" ++ unwords users
   40 
   41 -- |This function parses the messages from the client.
   42 parseMsg :: String      -- ^Message from the client
   43          -> Msg         -- ^Parsed message
   44 parseMsg str = parseCommand cmd after
   45         where (_, cmd, after) = str =~ "^[A-Z]+ " :: (String, String, String)
   46 
   47 -- |This function parses the additional information from the message.
   48 parseCommand :: String  -- ^Type of the message
   49              -> String  -- ^Remaining message
   50              -> Msg     -- ^Parsed message
   51 parseCommand "NICK " user = Nick user
   52 parseCommand "USER " after = User user $ tail name
   53         where (before, name, _) = after =~ ":.*$" :: (String, String, String)
   54               user = before =~ "^\\S+"
   55 parseCommand "PRIVMSG " after = PrivMsg (head groups) msg
   56         where (_, _, msg, groups) = after =~ "^(\\S+)\\s:" :: (String, String, String, [String])
   57 parseCommand "JOIN " channel = JoinMulti $ splitOn "," channel
   58 parseCommand "PART " after
   59         | after =~ "^(\\S+)\\s:" = Part (head groups) msg
   60         | otherwise = Part after ""
   61         where (_, _, msg, groups) = after =~ "^(\\S+)\\s:" :: (String, String, String, [String])
   62 parseCommand "PING " server = Ping server
   63 parseCommand "QUIT " after = Quit msg
   64         where msg = tail after
   65 parseCommand "ISON " after = IsOn $ words after
   66 parseCommand cmd after = Plain (cmd ++ after)
